## 1. 전체 진행 현황

> 각 정리노트의 frontmatter(`status`, `confidence`, `next_review`)를 기준으로 자동 정렬됩니다.

```dataview
TABLE 
  status AS "상태",
  confidence AS "자신감",
  round AS "회독수",
  next_review AS "다음 복습일",
  length(file.inlinks) AS "연결된 오답수"
FROM "CPA기업법1"
WHERE type = "정리노트"
SORT next_review ASC
```

## 8. 마인드맵 내보내기

> 현재 폴더의 정리노트를 **마인드맵 PNG/PDF**로 내보냅니다.

```dataviewjs
const folder = dv.current().file.folder;
const files = dv.pages(`"${folder}"`).where(p => p.file.name !== "00_대시보드").sort(p => p.file.name);

const nodeCmd = "/Users/a1/.nvm/versions/node/v20.19.0/bin/node";
const vaultPath = app.vault.adapter.basePath;
const scriptPath = `${vaultPath}/scripts/mindmap-export.js`;

const container = dv.container;

// 전체 내보내기 버튼
const headerDiv = container.createEl("div", { cls: "mindmap-header" });
headerDiv.style.marginBottom = "12px";

const btnAllPng = headerDiv.createEl("button", { text: "📸 전체 PNG 내보내기" });
btnAllPng.style.cssText = "margin-right:8px; padding:6px 14px; background:#4CAF50; color:white; border:none; border-radius:6px; cursor:pointer; font-size:14px;";

const btnAllPdf = headerDiv.createEl("button", { text: "📄 전체 PDF 내보내기" });
btnAllPdf.style.cssText = "padding:6px 14px; background:#2196F3; color:white; border:none; border-radius:6px; cursor:pointer; font-size:14px;";

async function runExport(filePath, format) {
  const { exec } = require('child_process');
  return new Promise((resolve, reject) => {
    exec(`"${nodeCmd}" "${scriptPath}" "${filePath}" "${format}"`,
      { timeout: 30000, cwd: `${vaultPath}/scripts` },
      (err, stdout, stderr) => {
        if (err) reject(err.message);
        else resolve(stdout.trim());
      });
  });
}

btnAllPng.onclick = async () => {
  new Notice("전체 PNG 내보내기 시작...");
  for (const p of files) {
    try {
      const fp = `${vaultPath}/${p.file.path}`;
      await runExport(fp, "png");
    } catch(e) { new Notice(`실패: ${p.file.name} - ${e}`, 3000); }
  }
  new Notice("✅ 전체 PNG 내보내기 완료!", 5000);
};

btnAllPdf.onclick = async () => {
  new Notice("전체 PDF 내보내기 시작...");
  for (const p of files) {
    try {
      const fp = `${vaultPath}/${p.file.path}`;
      await runExport(fp, "pdf");
    } catch(e) { new Notice(`실패: ${p.file.name} - ${e}`, 3000); }
  }
  new Notice("✅ 전체 PDF 내보내기 완료!", 5000);
};

// 개별 파일 버튼 테이블
const table = container.createEl("table");
table.style.cssText = "width:100%; border-collapse:collapse; margin-top:8px;";

const thead = table.createEl("thead");
const hr = thead.createEl("tr");
["노트", "PNG", "PDF"].forEach(h => {
  const th = hr.createEl("th", { text: h });
  th.style.cssText = "text-align:left; padding:6px 10px; border-bottom:2px solid var(--background-modifier-border);";
});

const tbody = table.createEl("tbody");
for (const p of files) {
  const tr = tbody.createEl("tr");

  const tdName = tr.createEl("td");
  tdName.style.cssText = "padding:5px 10px; border-bottom:1px solid var(--background-modifier-border);";
  const link = tdName.createEl("a", { text: p.file.name, href: p.file.path });
  link.classList.add("internal-link");

  const tdPng = tr.createEl("td");
  tdPng.style.cssText = "padding:5px 10px; border-bottom:1px solid var(--background-modifier-border);";
  const bPng = tdPng.createEl("button", { text: "📸 PNG" });
  bPng.style.cssText = "padding:3px 10px; background:#4CAF50; color:white; border:none; border-radius:4px; cursor:pointer; font-size:12px;";
  bPng.onclick = async () => {
    new Notice(`${p.file.name} PNG 내보내기 중...`);
    try {
      const result = await runExport(`${vaultPath}/${p.file.path}`, "png");
      new Notice(`✅ ${p.file.name} PNG 완료!\n${result}`, 5000);
    } catch(e) { new Notice(`❌ 실패: ${e}`, 5000); }
  };

  const tdPdf = tr.createEl("td");
  tdPdf.style.cssText = "padding:5px 10px; border-bottom:1px solid var(--background-modifier-border);";
  const bPdf = tdPdf.createEl("button", { text: "📄 PDF" });
  bPdf.style.cssText = "padding:3px 10px; background:#2196F3; color:white; border:none; border-radius:4px; cursor:pointer; font-size:12px;";
  bPdf.onclick = async () => {
    new Notice(`${p.file.name} PDF 내보내기 중...`);
    try {
      const result = await runExport(`${vaultPath}/${p.file.path}`, "pdf");
      new Notice(`✅ ${p.file.name} PDF 완료!\n${result}`, 5000);
    } catch(e) { new Notice(`❌ 실패: ${e}`, 5000); }
  };
}
```

## 📌 사용법 요약

| 할 일    | 방법                                                  |
| ------ | --------------------------------------------------- |
| 공부 시작  | Daily Note 생성 → `daily_log` 템플릿 적용                  |
| 문제 틀림  | 새 노트 → `오답노트` 템플릿 → `오답노트/` 폴더에 저장                  |
| 백지복원   | 새 노트 → `백지복원` 템플릿 → 기억나는 대로 작성                      |
| 회독 완료  | 해당 정리노트 frontmatter에서 `round` +1, `status` 업데이트     |
| 약점 발견  | 정리노트 frontmatter `weak_points`에 추가, `confidence` 조정 |
| 복습일 도래 | 이 대시보드의 "오늘 복습해야 할 노트" 확인                           |
