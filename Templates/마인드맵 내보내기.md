const nodeCmd = "/Users/a1/.nvm/versions/node/v20.19.0/bin/node";
const vaultPath = app.vault.adapter.basePath;
const scriptPath = `${vaultPath}/scripts/mindmap-export.js`;
const file = tp.config.active_file;
const filePath = `${vaultPath}/${file.path}`;

const format = await tp.system.suggester(
  ["📸 PNG으로 내보내기", "📄 PDF로 내보내기", "📸📄 PNG + PDF 모두"],
  ["png", "pdf", "both"],

if (format) {
  new Notice(`마인드맵 ${format.toUpperCase()} 내보내기 중...`);
  try {
    const result = await tp.user.mindmap_export(format);
    new Notice(`✅ 마인드맵 내보내기 완료!\n${result}`, 5000);
  } catch(e) {
    // fallback: system command
    const { exec } = require('child_process');
    await new Promise((resolve, reject) => {
      exec(`"${nodeCmd}" "${scriptPath}" "${filePath}" "${format}"`,
        { timeout: 30000, cwd: `${vaultPath}/scripts` },
        (err, stdout, stderr) => {
          if (err) {
            new Notice(`❌ 실패: ${err.message}`, 5000);
            reject(err);
          } else {
            new Notice(`✅ 마인드맵 내보내기 완료!\n${stdout.trim()}`, 5000);
            resolve(stdout);
          }
        });
    });
  }
}
